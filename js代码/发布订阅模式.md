#发布订阅模式
```javascript
    let _event = {
        list : {},
        //绑定事件
        on ( key, fn ){
            if( !this.list[key] ){
                this.list[key] = [];
            }
            this.list[key].push(fn)
        },
        //触发事件
        emit (){
            let key = [].shift.call(arguments),
                fns = this.list[key];

            if ( !fns || fns.length === 0 ) { return false; }
            
            fns.forEach( (ele, index) => {
                ele.apply(this, arguments);
            } )
        },
        //删除 事件
        remove(key, fn){
            let fns = this.list[key];
            if( !fns ) { return false; }
            if( !fn ){
                fns && ( fns.length = 0 );
            } else {
                fns.forEach( ( ele, index )=>{
                    if ( ele === fn ){
                        fns.splice(index ,1);
                    }
                })
            }
        }
    }
```
